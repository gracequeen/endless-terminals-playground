def compute_integrity_hash(record: Dict[str, Any]) -> str:
    """Compute SHA-256 hash of canonical form."""
    amount_formatted = f"{float(record['amount']):.2f}"
    canonical = f"{record['transaction_id']}|{record['timestamp']}|{record['account_id']}|{amount_formatted}|{record['currency']}|{record['status']}"
    return hashlib.sha256(canonical.encode('utf-8')).hexdigest()
