def get_package_version(package):
    """Get installed version of package"""
    output = run_command(f"dpkg-query -W -f='${{Version}}' {package} 2>/dev/null")
    version = output.strip()
    if version:
        return version
    return "unknown"

def has_security_update(package):
    """Check if package has security update available"""
    output = run_command(f"apt-cache policy {package} 2>/dev/null")
    lines = output.split('\n')

    installed_version = None
    candidate_version = None
    has_security = False

    for line in lines:
        if 'Installed:' in line:
            installed_version = line.split(':', 1)[1].strip()
        elif 'Candidate:' in line:
            candidate_version = line.split(':', 1)[1].strip()
        elif 'security' in line.lower():
            has_security = True

    # If candidate version is different and there's a security repo, likely has update
    if installed_version and candidate_version:
        if installed_version != candidate_version and installed_version != '(none)':
            return True
        if has_security and candidate_version != '(none)':
            return True

    return False
