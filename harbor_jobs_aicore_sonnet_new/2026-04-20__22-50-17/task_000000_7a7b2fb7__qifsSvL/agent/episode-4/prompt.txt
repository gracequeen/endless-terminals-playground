def has_security_update(package):
    """Check if a package has a security update available."""
    try:
        result = subprocess.run(
            ['apt-cache', 'policy', package],
            capture_output=True, text=True, timeout=5
        )
        output = result.stdout

        # Look for security in the output
        if '-security' in output.lower() or 'security.ubuntu.com' in output.lower():
            # Check if there's a candidate version different from installed
            installed_ver = None
            candidate_ver = None
            for line in output.split('\n'):
                if 'Installed:' in line:
                    installed_ver = line.split(':', 1)[1].strip()
                elif 'Candidate:' in line:
                    candidate_ver = line.split(':', 1)[1].strip()

            if installed_ver and candidate_ver and installed_ver != candidate_ver and installed_ver != '(none)':
                return True
        return False
    except Exception:
        return False
