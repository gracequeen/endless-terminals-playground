def check_connection(src_ip, dest_port, rules, default_action):
    """Check if a connection is allowed by firewall rules."""
    src_ip_obj = ipaddress.ip_address(src_ip)
    dest_port_int = int(dest_port)

    # Find the most specific matching rule
    for rule in rules:
        if src_ip_obj in rule['network']:
            # Check if port is allowed
            if dest_port_int in rule['allowed_ports']:
                if rule['action'] == 'allow':
                    return True, None
                else:  # explicit deny
                    return False, rule['source_cidr']
            # If port doesn't match, continue to less specific rules

    # No matching rule found, apply default action
    return False, None
