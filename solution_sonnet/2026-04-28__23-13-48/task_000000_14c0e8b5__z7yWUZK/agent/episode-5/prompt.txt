def parse_port_spec(port_spec):
    """Parse a port specification into a list of allowed ports or a range."""
    port_str = str(port_spec)
    if '-' in port_str:
        start, end = port_str.split('-')
        return ('range', int(start), int(end))
    else:
        return ('single', int(port_str))

def port_matches(port, port_spec):
    """Check if a port matches a port specification."""
    spec_type, *spec_data = parse_port_spec(port_spec)
    if spec_type == 'single':
        return port == spec_data[0]
