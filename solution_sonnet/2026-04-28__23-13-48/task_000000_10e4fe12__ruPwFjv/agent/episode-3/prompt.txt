#!/bin/bash

# Backup script that copies accessible files from /home/user/projects to /home/user/backups
# and creates a manifest with ownership information

SOURCE_DIR="/home/user/projects"
BACKUP_DIR="/home/user/backups"
MANIFEST="$BACKUP_DIR/manifest.txt"

# Create backup directory if it doesn't exist
mkdir -p "$BACKUP_DIR"

# Clear or create manifest file
> "$MANIFEST"

echo "Starting backup from $SOURCE_DIR to $BACKUP_DIR..."

# Find all files (not directories) in the source directory
while IFS= read -r -d '' file; do
    # Get relative path
    rel_path="${file#$SOURCE_DIR/}"
    dest_file="$BACKUP_DIR/$rel_path"

    # Check if file is readable
    if [ -r "$file" ]; then
        # Get owner:group information
        owner_info=$(stat -c "%U:%G" "$file" 2>/dev/null)

        # Create destination directory if needed
        dest_dir=$(dirname "$dest_file")
        mkdir -p "$dest_dir" 2>/dev/null

        # Copy the file
        if cp "$file" "$dest_file" 2>/dev/null; then
            # Record in manifest
            echo "$rel_path $owner_info" >> "$MANIFEST"
            echo "Copied: $rel_path (owner: $owner_info)"
        fi
    fi
done < <(find "$SOURCE_DIR" -type f -print0 2>/dev/null)

echo "Backup complete. Manifest written to $MANIFEST"
