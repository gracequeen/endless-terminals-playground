#!/bin/bash

# Backup script that copies accessible files from /home/user/projects
# to /home/user/backups while preserving directory structure

SOURCE_DIR="/home/user/projects"
BACKUP_DIR="/home/user/backups"
MANIFEST="${BACKUP_DIR}/manifest.txt"

# Clear or create manifest
> "$MANIFEST"

# Find all files we can access (suppress permission errors)
find "$SOURCE_DIR" -type f 2>/dev/null | while IFS= read -r file; do
    # Try to read the file's ownership info
    if owner_info=$(stat -c "%U:%G" "$file" 2>/dev/null); then
        # Calculate relative path
        rel_path="${file#$SOURCE_DIR/}"
        dest_file="${BACKUP_DIR}/${rel_path}"
        dest_dir=$(dirname "$dest_file")

        # Create destination directory structure
        mkdir -p "$dest_dir" 2>/dev/null

        # Try to copy the file
        if cp "$file" "$dest_file" 2>/dev/null; then
            # Record in manifest
            echo "${rel_path} ${owner_info}" >> "$MANIFEST"
        fi
    fi
done

echo "Backup complete. Manifest written to $MANIFEST"
