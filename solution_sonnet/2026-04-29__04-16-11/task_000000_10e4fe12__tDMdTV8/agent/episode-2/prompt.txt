#!/bin/bash

# Backup script that copies accessible files from /home/user/projects
# to /home/user/backups while preserving directory structure

SOURCE_DIR="/home/user/projects"
BACKUP_DIR="/home/user/backups"
MANIFEST="${BACKUP_DIR}/manifest.txt"

# Clear/create manifest file
> "$MANIFEST"

# Find all accessible files and process them
find "$SOURCE_DIR" -type f 2>/dev/null | while read -r file; do
    # Try to read the file (test if we have read access)
    if [ -r "$file" ]; then
        # Get relative path from source directory
        rel_path="${file#$SOURCE_DIR/}"
        dest_file="${BACKUP_DIR}/${rel_path}"
        dest_dir="$(dirname "$dest_file")"

        # Create destination directory structure
        mkdir -p "$dest_dir" 2>/dev/null

        # Copy the file
        if cp "$file" "$dest_file" 2>/dev/null; then
            # Get owner:group information
            owner_group=$(stat -c "%U:%G" "$file" 2>/dev/null)

            # Write to manifest
            echo "${rel_path} ${owner_group}" >> "$MANIFEST"
        fi
    fi
done

echo "Backup completed. Manifest written to $MANIFEST"
