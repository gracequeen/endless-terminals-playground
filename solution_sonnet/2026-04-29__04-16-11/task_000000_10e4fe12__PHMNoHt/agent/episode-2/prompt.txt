#!/bin/bash

# Backup script that handles permission issues gracefully
SOURCE="/home/user/projects"
BACKUP="/home/user/backups"
MANIFEST="$BACKUP/manifest.txt"

# Clear/create manifest
> "$MANIFEST"

echo "Starting backup from $SOURCE to $BACKUP..."

# Find all files, suppressing permission errors
find "$SOURCE" -type f 2>/dev/null | while IFS= read -r file; do
    # Check if file is readable
    if [ -r "$file" ]; then
        # Get relative path
        rel_path="${file#$SOURCE/}"
        dest_file="$BACKUP/$rel_path"
        dest_dir="$(dirname "$dest_file")"

        # Create destination directory if needed
        mkdir -p "$dest_dir" 2>/dev/null

        # Copy the file
        if cp "$file" "$dest_file" 2>/dev/null; then
            # Get owner:group info using stat
            owner_info=$(stat -c "%U:%G" "$file" 2>/dev/null)
            if [ -z "$owner_info" ]; then
                owner_info="unknown:unknown"
            fi

            # Write to manifest
            echo "$rel_path $owner_info" >> "$MANIFEST"
            echo "Copied: $rel_path (owner: $owner_info)"
        fi
    fi
done

echo "Backup complete. Manifest written to $MANIFEST"
